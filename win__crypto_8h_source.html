249  std::error_code& ec) {
250  ec.clear();
251 
252  DWORD dwBufferLen = 0;
253  if (!CryptStringToBinaryA(certstr.data(), certstr.size(), CRYPT_STRING_BASE64HEADER, NULL, &dwBufferLen,
254  NULL, NULL)) {
255  ec = error::rsa_error::cert_load_failed;
256  return {};
257  }
258 
259  std::vector<BYTE> buffer(dwBufferLen);
260  if (!CryptStringToBinaryA(certstr.data(), certstr.size(), CRYPT_STRING_BASE64HEADER, buffer.data(),
261  &dwBufferLen, NULL, NULL)) {
262  ec = error::rsa_error::cert_load_failed;
263  return {};
264  }
265 
266  DWORD cbSignedContentInfoBuffer = 0;
267  LocalBuffer<BYTE> signedContentInfoBuffer;
268  if (!CryptDecodeObjectEx(X509_ASN_ENCODING | PKCS_7_ASN_ENCODING, X509_CERT, buffer.data(), dwBufferLen,
269  CRYPT_DECODE_ALLOC_FLAG | CRYPT_DECODE_NOCOPY_FLAG, NULL,
270  &signedContentInfoBuffer, &cbSignedContentInfoBuffer)) {
271  ec = error::rsa_error::cert_load_failed;
272  return {};
273  }
274 
275  auto pSignedContentInfo = reinterpret_cast<PCERT_SIGNED_CONTENT_INFO>(signedContentInfoBuffer.get());
276 
277  CCRYPT_OID_INFO* oidInfo =
278  CryptFindOIDInfo(CRYPT_OID_INFO_OID_KEY, pSignedContentInfo->SignatureAlgorithm.pszObjId, 0);
279  if (!oidInfo || oidInfo->dwGroupId != CRYPT_SIGN_ALG_OID_GROUP_ID) {
280  ec = error::rsa_error::cert_load_failed;
281  return {};
282  }
283 
284  DWORD cbCertInfoBuffer = 0;
285  LocalBuffer<BYTE> certInfoBuffer;
286  if (!CryptDecodeObjectEx(X509_ASN_ENCODING, X509_CERT_TO_BE_SIGNED,
287  pSignedContentInfo->ToBeSigned.pbData, pSignedContentInfo->ToBeSigned.cbData,
288  CRYPT_DECODE_ALLOC_FLAG | CRYPT_DECODE_NOCOPY_FLAG, NULL, &certInfoBuffer,
289  &cbCertInfoBuffer)) {
290  ec = error::rsa_error::cert_load_failed;
291  return {};
292  }
293 
294  auto pCertInfo = reinterpret_cast<PCERT_INFO>(certInfoBuffer.get());
295 
296  CCryptContext context;
297  if (!CryptAcquireContext(&context, NULL, NULL, PROV_RSA_FULL, CRYPT_VERIFYCONTEXT | CRYPT_SILENT)) {
298  ec = error::rsa_error::cert_load_failed;
299  return {};
300  }
301 
302  CCryptKey key;
303  if (!CryptImportPublicKeyInfo(context.get(), X509_ASN_ENCODING, &pCertInfo->SubjectPublicKeyInfo,
304  &key)) {
305  ec = error::rsa_error::cert_load_failed;
306  return {};
307  }
308

311  ec = error::rsa_error::cert_load_failed;
312  return {};
