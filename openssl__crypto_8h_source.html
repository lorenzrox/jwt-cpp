863  void verify(const std::string& data, const std::string& signature, std::error_code& ec) const {
864  ec.clear();
865  auto hash = this->generate_hash(data, ec);
866  if (ec) return;
867 
868  std::unique_ptr<RSA, decltype(&RSA_free)> key(EVP_PKEY_get1_RSA(pkey.get()), RSA_free);
869  if (!key) {
870  ec = error::signature_verification_error::get_key_failed;
871  return;
872  }
873  const int size = RSA_size(key.get());
874 
875  std::string sig(size, 0x00);
876  if (RSA_public_decrypt(
877  static_cast<int>(signature.size()),
878  reinterpret_cast<const unsigned char*>(signature.data()),
879  (unsigned char*)sig.data(), // NOLINT(google-readability-casting) requires `const_cast`
880  key.get(), RSA_NO_PADDING) == 0) {
881  ec = error::signature_verification_error::invalid_signature;
882  return;
883  }
884 
885  if (RSA_verify_PKCS1_PSS_mgf1(key.get(), reinterpret_cast<const unsigned char*>(hash.data()), md(),
886  md(), reinterpret_cast<const unsigned char*>(sig.data()), -1) == 0) {
887  ec = error::signature_verification_error::invalid_signature;
888  return;
889  }
890  }

904  std::string generate_hash(const std::string& data, std::error_code& ec) const {
905 #ifdef OPENSSL10
906  std::unique_ptr<EVP_MD_CTX, decltype(&EVP_MD_CTX_destroy)> ctx(EVP_MD_CTX_create(),
907  &EVP_MD_CTX_destroy);
908 #else
909  std::unique_ptr<EVP_MD_CTX, decltype(&EVP_MD_CTX_free)> ctx(EVP_MD_CTX_new(), EVP_MD_CTX_free);
910 #endif
911  if (!ctx) {
912  ec = error::signature_generation_error::create_context_failed;
913  return {};
914  }
915  if (EVP_DigestInit(ctx.get(), md()) == 0) {
916  ec = error::signature_generation_error::digestinit_failed;
917  return {};
918  }
919  if (EVP_DigestUpdate(ctx.get(), data.data(), data.size()) == 0) {
920  ec = error::signature_generation_error::digestupdate_failed;
921  return {};
922  }
923  unsigned int len = 0;
924  std::string res(EVP_MD_CTX_size(ctx.get()), '\0');
925  if (EVP_DigestFinal(ctx.get(), (unsigned char*)res.data(), &len) ==
926  0) { // NOLINT(google-readability-casting) requires `const_cast`
927  ec = error::signature_generation_error::digestfinal_failed;
928  return {};
929  }
930  res.resize(len);
931  return res;
